Build comprehensive marketing and sales tools to help CPP/Filta grow their customer base and close more deals.

**IMPORTANT:**
- Build HoodOps pages in `apps/web-vendor/` (the HoodOps vendor app)
- Deploy HoodOps to web-vendor.vercel.app
- Company name is "HoodOps" (not "HoodOps PRO")
- NO demo/fake data - use real API hooks with empty states

- All cards center-aligned

## Part 1: Sales Pipeline & CRM

### 1.1 Database Schema

```sql
-- Leads table
CREATE TABLE leads (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vendor_id UUID NOT NULL REFERENCES vendors(id),
  
  -- Lead info
  business_name TEXT NOT NULL,
  contact_name TEXT,
  contact_email TEXT,
  contact_phone TEXT,
  address TEXT,
  city TEXT,
  state TEXT,
  zip TEXT,
  
  -- Classification
  lead_source TEXT NOT NULL, -- 'ai_estimate', 'referral', 'website', 'cold_call', 'door_knock', 'violation_outreach', 'google_ads', 'facebook', 'yelp', 'thumbtack', 'homeadvisor', 'trade_show', 'other'
  lead_type TEXT DEFAULT 'prospect', -- 'prospect', 'qualified', 'opportunity', 'customer', 'lost'
  industry TEXT, -- 'restaurant', 'hotel', 'hospital', 'school', 'catering', 'food_truck', 'other'
  kitchen_type TEXT, -- 'full_service', 'fast_food', 'ghost_kitchen', 'commissary', 'institutional'
  estimated_value DECIMAL(10,2),
  estimated_monthly_value DECIMAL(10,2),
  
  -- Pipeline
  pipeline_stage TEXT DEFAULT 'new', -- 'new', 'contacted', 'qualified', 'proposal_sent', 'negotiating', 'won', 'lost'
  probability INTEGER DEFAULT 10, -- Win probability %
  expected_close_date DATE,
  
  -- Assignment
  assigned_to UUID REFERENCES employees(id),
  
  -- Tracking
  last_contact_at TIMESTAMPTZ,
  next_followup_at TIMESTAMPTZ,
  followup_count INTEGER DEFAULT 0,
  
  -- If converted
  organization_id UUID REFERENCES organizations(id),
  first_job_id UUID REFERENCES jobs(id),
  converted_at TIMESTAMPTZ,
  
  -- If lost
  lost_reason TEXT,
  lost_at TIMESTAMPTZ,
  
  -- Source tracking
  referral_id UUID REFERENCES referrals(id),
  service_request_id UUID REFERENCES service_requests(id),
  campaign_id UUID REFERENCES marketing_campaigns(id),
  utm_source TEXT,
  utm_medium TEXT,
  utm_campaign TEXT,
  
  notes TEXT,
  tags TEXT[],
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_leads_stage ON leads(pipeline_stage);
CREATE INDEX idx_leads_assigned ON leads(assigned_to);
CREATE INDEX idx_leads_source ON leads(lead_source);

-- Lead activities
CREATE TABLE lead_activities (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vendor_id UUID NOT NULL REFERENCES vendors(id),
  lead_id UUID NOT NULL REFERENCES leads(id) ON DELETE CASCADE,
  
  activity_type TEXT NOT NULL, -- 'call', 'email', 'meeting', 'site_visit', 'estimate_sent', 'proposal_sent', 'follow_up', 'note', 'stage_change'
  description TEXT,
  outcome TEXT, -- 'connected', 'voicemail', 'no_answer', 'scheduled', 'not_interested', 'call_back'
  
  -- For calls
  call_duration_seconds INTEGER,
  
  -- For emails
  email_subject TEXT,
  email_opened BOOLEAN,
  email_clicked BOOLEAN,
  
  -- For meetings/site visits
  meeting_date TIMESTAMPTZ,
  meeting_location TEXT,
  
  performed_by UUID REFERENCES employees(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Sales goals
CREATE TABLE sales_goals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vendor_id UUID NOT NULL REFERENCES vendors(id),
  employee_id UUID REFERENCES employees(id), -- NULL = company-wide
  
  period_type TEXT NOT NULL, -- 'monthly', 'quarterly', 'yearly'
  period_start DATE NOT NULL,
  period_end DATE NOT NULL,
  
  -- Targets
  revenue_target DECIMAL(12,2),
  new_customers_target INTEGER,
  leads_target INTEGER,
  jobs_target INTEGER,
  
  -- Actual
  revenue_actual DECIMAL(12,2) DEFAULT 0,
  new_customers_actual INTEGER DEFAULT 0,
  leads_actual INTEGER DEFAULT 0,
  jobs_actual INTEGER DEFAULT 0,
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

  address TEXT,
  city TEXT,
  state TEXT,
  zip TEXT,
  phone TEXT,
  email TEXT,

  status TEXT DEFAULT 'pending', -- 'pending', 'contacted', 'interested', 'not_interested', 'converted', 'invalid'

  last_outreach_at TIMESTAMPTZ,

  lead_id UUID REFERENCES leads(id),

### 3.1 Database Schema

```sql
CREATE TABLE outreach_sequences (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vendor_id UUID NOT NULL REFERENCES vendors(id),
  
  name TEXT NOT NULL,
  description TEXT,
  trigger_type TEXT NOT NULL, -- 'new_lead', 'estimate_sent', 'job_complete', 'no_response', 'lapsed_customer', 'violation_found'
  
  is_active BOOLEAN DEFAULT true,
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE outreach_sequence_steps (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  sequence_id UUID NOT NULL REFERENCES outreach_sequences(id) ON DELETE CASCADE,
  
  step_order INTEGER NOT NULL,
  delay_days INTEGER DEFAULT 0, -- Days after trigger or previous step
  delay_hours INTEGER DEFAULT 0,
  
  action_type TEXT NOT NULL, -- 'email', 'sms', 'task', 'call_reminder'
  template_id UUID,
  subject TEXT,
  message TEXT,
  
  -- Conditions
  skip_if TEXT, -- 'responded', 'converted', 'unsubscribed'
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE outreach_enrollments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  sequence_id UUID NOT NULL REFERENCES outreach_sequences(id),
  lead_id UUID REFERENCES leads(id),
  organization_id UUID REFERENCES organizations(id),
  
  status TEXT DEFAULT 'active', -- 'active', 'completed', 'stopped', 'converted'
  current_step INTEGER DEFAULT 0,
  
  enrolled_at TIMESTAMPTZ DEFAULT NOW(),
  completed_at TIMESTAMPTZ,
  stopped_reason TEXT
);
```

## Part 5: Service Agreements

### 5.1 Database Schema

```sql
-- Service agreement templates
CREATE TABLE agreement_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vendor_id UUID NOT NULL REFERENCES vendors(id),
  
  name TEXT NOT NULL,
  description TEXT,
  
  -- Template content
  content_html TEXT NOT NULL, -- Rich text with merge fields
  content_json JSONB, -- Structured sections for editor
  
  -- Default terms
  default_term_months INTEGER DEFAULT 12,
  default_payment_terms TEXT DEFAULT 'net_30', -- 'due_on_receipt', 'net_15', 'net_30', 'net_60'
  default_auto_renew BOOLEAN DEFAULT true,
  default_cancellation_days INTEGER DEFAULT 30,
  
  -- Services included
  default_services JSONB, -- [{service_type, frequency, price}]
  
  -- Legal
  terms_and_conditions TEXT,
  liability_limit DECIMAL(12,2),
  insurance_requirements TEXT,
  
  is_default BOOLEAN DEFAULT false,
  is_active BOOLEAN DEFAULT true,
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Service agreements (contracts)
CREATE TABLE service_agreements (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vendor_id UUID NOT NULL REFERENCES vendors(id),
  template_id UUID REFERENCES agreement_templates(id),
  
  -- Agreement number
  agreement_number TEXT NOT NULL UNIQUE,
  
  -- Parties
  organization_id UUID NOT NULL REFERENCES organizations(id),
  location_ids UUID[], -- Which locations covered
  
  -- Contact who signed
  signer_name TEXT,
  signer_title TEXT,
  signer_email TEXT,
  
  -- Term
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  term_months INTEGER NOT NULL,
  auto_renew BOOLEAN DEFAULT true,
  cancellation_notice_days INTEGER DEFAULT 30,
  
  -- Pricing
  pricing_type TEXT DEFAULT 'per_service', -- 'per_service', 'monthly_flat', 'annual_flat', 'custom'
  monthly_amount DECIMAL(10,2),
  annual_amount DECIMAL(10,2),
  
  -- Services included
  services JSONB NOT NULL, -- [{service_type, frequency, locations, price_per_service}]
  
  -- Payment
  payment_terms TEXT DEFAULT 'net_30',
  billing_frequency TEXT DEFAULT 'per_service', -- 'per_service', 'monthly', 'quarterly', 'annually'
  
  -- Discounts
  discount_percent DECIMAL(5,2) DEFAULT 0,
  discount_reason TEXT,
  
  -- Status
  status TEXT DEFAULT 'draft', -- 'draft', 'sent', 'viewed', 'signed', 'active', 'expired', 'cancelled', 'renewed'
  
  -- Signature tracking
  sent_at TIMESTAMPTZ,
  viewed_at TIMESTAMPTZ,
  signed_at TIMESTAMPTZ,
  signature_ip TEXT,
  signature_data JSONB, -- Signature image or typed name
  
  -- Document
  pdf_url TEXT,
  signed_pdf_url TEXT,
  
  -- Renewal
  renewed_from_id UUID REFERENCES service_agreements(id),
  renewed_to_id UUID REFERENCES service_agreements(id),
  renewal_reminder_sent BOOLEAN DEFAULT false,
  
  -- Internal
  created_by UUID REFERENCES employees(id),
  notes TEXT,
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_agreements_org ON service_agreements(organization_id);
CREATE INDEX idx_agreements_status ON service_agreements(status);
CREATE INDEX idx_agreements_dates ON service_agreements(start_date, end_date);

-- Agreement activity log
CREATE TABLE agreement_activities (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  agreement_id UUID NOT NULL REFERENCES service_agreements(id) ON DELETE CASCADE,
  
  activity_type TEXT NOT NULL, -- 'created', 'sent', 'viewed', 'signed', 'reminder_sent', 'renewed', 'cancelled', 'modified'
  description TEXT,
  performed_by UUID REFERENCES employees(id),
  ip_address TEXT,
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```
